13 /// @brief A fixed-grid bitmap font: one atlas texture holding every
14 /// character in a regular grid, in ascii order.
15 ///
16 /// All glyph data lives in the atlas image (content/fonts/), this struct
17 /// only describes the grid so draw_text can find each character:
18 /// @code{.cpp}
19 /// dyro::font font;
20 /// font.atlas = engine.get_texture_loader().load_from_file(path_to_atlas);
21 /// renderer.draw_text(font, "hello", { 20.0f, 20.0f }, 16.0f);
22 /// @endcode
23 ///
24 /// The defaults match the engine's built-in atlas
25 /// (content/fonts/font_8x8.png): 8x8 pixel glyphs, 16 per row, starting
26 /// at the space character (ascii 32).
27 struct font
28 {
29 std::shared_ptr<texture> atlas;
30
31 uint32_t first_character = 32;
32 uint32_t glyph_count = 96;
33 uint32_t glyphs_per_row = 16;
34 uvec2 glyph_size = { 8, 8 };
35
36 //----------------------------------------------------------
37 /// @brief Returns where a character's glyph sits in the atlas, in
38 /// pixels. Characters the font does not have come out as the first
39 /// glyph (usually space, so they show up as blank).
40 rect get_glyph_source_rect(char character) const
41 {
42 // The cast via unsigned char keeps characters above 127 from
43 // going negative and ending up as a huge index.
44 uint32_t index = static_cast<unsigned char>(character) - first_character;
45 if (index >= glyph_count)
46 {
47 index = 0;
48 }
49
50 const vec2 top_left = {
51 static_cast<float>((index % glyphs_per_row) * glyph_size.x),
52 static_cast<float>((index / glyphs_per_row) * glyph_size.y) };
53
54 return { top_left, top_left + vec2(glyph_size) };
55 }
56 };
