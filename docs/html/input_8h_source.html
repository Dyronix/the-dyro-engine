49 /// @brief Keeps track of the keyboard and mouse state.
50 ///
51 /// The engine owns one instance (engine::get_input) and the window feeds
52 /// it every input message. Game code just asks questions:
53 /// @code{.cpp}
54 /// if (input.is_key_down(dyx::key::a)) { } // every frame while held
55 /// if (input.was_key_pressed(dyx::key::space)) { } // only the frame it went down
56 /// if (input.was_key_released(dyx::key::escape)) { } // only the frame it came up
57 /// @endcode
58 class input
59 {
60 public:
61 //----------------------------------------------------------
62 /// @brief Returns true while the key is held down.
63 bool is_key_down(key value) const { return m_current_keys[static_cast<uint8_t>(value)]; }
64
65 //----------------------------------------------------------
66 /// @brief Returns true only in the frame the key went down.
67 bool was_key_pressed(key value) const
68 {
69 return m_current_keys[static_cast<uint8_t>(value)] && !m_previous_keys[static_cast<uint8_t>(value)];
70 }
71
72 //----------------------------------------------------------
73 /// @brief Returns true only in the frame the key came up.
74 bool was_key_released(key value) const
75 {
76 return !m_current_keys[static_cast<uint8_t>(value)] && m_previous_keys[static_cast<uint8_t>(value)];
77 }
78
79 //----------------------------------------------------------
80 /// @brief Returns true while the mouse button is held down.

84 /// @brief Returns true only in the frame the mouse button went down.
