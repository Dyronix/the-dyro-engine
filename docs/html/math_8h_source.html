59 /// @brief Builds an orthographic projection for 2D rendering.
60 ///
61 /// Maps pixel coordinates to normalized device coordinates, with the
62 /// origin (0, 0) in the top-left corner of the screen and the y axis
63 /// pointing down (like most 2D drawing APIs).
64 ///
65 /// Note on conventions: the engine follows glm, so matrices are stored
66 /// column-major (m[column][row]) and positions are column vectors that
67 /// are transformed as v' = M * v. The translation lives in the last
68 /// column of the matrix.
69 ///
70 /// @param width Width of the render area in pixels.
71 /// @param height Height of the render area in pixels.
72 /// @return Orthographic projection matrix.
73 inline mat4 make_orthographic_projection(float width, float height)
74 {
75 mat4 projection(0.0f);
76 projection[0][0] = 2.0f / width; // x: [0, width] -> [-1, 1]
77 projection[1][1] = -2.0f / height; // y: [0, height] -> [1, -1] (flip: screen y is down, clip space y is up)
78 projection[2][2] = 1.0f;
79 projection[3][0] = -1.0f;
80 projection[3][1] = 1.0f;
81 projection[3][3] = 1.0f;
82
83 return projection;
84 }

87 /// @brief Builds the world transformation for a sprite quad.
88 ///
89 /// The renderer draws a unit quad from (-0.5, -0.5) to (0.5, 0.5). This
90 /// transform scales it to the requested size, rotates it around its
91 /// center and then moves it so its center sits at the given position.
92 ///
93 /// @param position Position of the sprite center in pixels.
94 /// @param size Width and height of the sprite in pixels.
95 /// @param rotation_radians Clockwise rotation around the sprite center.
96 /// @return World transformation matrix for the sprite.
97 inline mat4 make_sprite_transform(vec2 position, vec2 size, float rotation_radians)
98 {
99 // Column-vector convention: the transform written last is applied
100 // first, so the quad is scaled, then rotated, then translated.
101 mat4 transform = glm::translate(mat4(1.0f), vec3(position, 0.0f));
102 transform = glm::rotate(transform, rotation_radians, vec3(0.0f, 0.0f, 1.0f));
103 transform = glm::scale(transform, vec3(size, 1.0f));
104
105 return transform;
106 }
